\begin{lem}
    The homogeneous cliques in a vertex-critical counterexample to BK have at most three vertices.
\end{lem}
\begin{proof}
    Suppose $G$ is a vertex-critical counterexample to BK.  
    Choose homogeneous clique $S\subseteq V(G)$ maximizing $\card{S}$.
    Put $T \DefinedAs N(S)\setminus S$. 
    By homogeneity, $S$ is joined to $T$.
    Plainly, $\card{S} + \card{T} \ge \Delta(G) + 1$.

    Suppose the lemma is false.
    Then $\card{S} \ge 4$. By Lemma 4.49 in 
    \href{https://arxiv.org/pdf/1203.5380v1.pdf}{original mules} 
    $\card{T} \le 3$.  By Lemma 4.29  in 
    \href{https://arxiv.org/pdf/1203.5380v1.pdf}{original mules} there is
    no induced $K_6 \vee E_3$, so we must have $\card{S} \le 5$. 
    But then $\Delta(G) + 1 \ge 10 > 8 \ge \card{S} + \card{T}$, a contradiction.


\end{proof}
